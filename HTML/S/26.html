  83         move_point.x = 1;

 119   }

 121     rect.x = 0;

 132   uint8_t used = character_use_pool->status->max_size - character_use_pool->status->current_size;

 148 Execute_Result init_view(SDL_Renderer_Access render) {
 149   SDL_Color white = {255, 255, 255};
 150 
 151   TTF_Font* font = TTF_OpenFont(FONT_FAMILY, 512);
 152   if (!font) {
 153     printf("TTF_OpenFont: %s\n", TTF_GetError());
 154     return EXECUTE_FAILED;
 155   }
 156 
 157   SDL_Surface* surfaceMessage = NULL;
 158   uint8_t counter = 0;
 159   Style_Access result = Style_Pool_Interface.next(style_pool, &counter);
 160 
 161   while (result != NULL) {
 162     surfaceMessage = TTF_RenderUTF8_Solid(font,
 163         result->mark,
 164         white);
 165     result->access = SDL_CreateTextureFromSurface(render, surfaceMessage);
 166 
 167     SDL_FreeSurface(surfaceMessage);
 168 
 169     result = Style_Pool_Interface.next(style_pool, &counter);
 170   }
 171   TTF_CloseFont(font);
 172 
 173   result = Style_Pool_Interface.find(style_pool, "player");
 174   character.set_style(Player, result);
 175 
 176   dead = Style_Pool_Interface.find(style_pool, "dead");
