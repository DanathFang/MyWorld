 100 void setup_npc(char *file_path) {

 102 
 103   mpc_parser_t* Number         = mpc_new("number");
 104   mpc_parser_t* Key            = mpc_new("key");
 105   mpc_parser_t* Json_String    = mpc_new("json_string");
 106   mpc_parser_t* Json_Boolean   = mpc_new("json_boolean");
 107   mpc_parser_t* Json_Value     = mpc_new("json_value");
 108   mpc_parser_t* KV_Item        = mpc_new("kv_item");
 109   mpc_parser_t* Json_Array     = mpc_new("json_array");
 110   mpc_parser_t* Json_Object    = mpc_new("json_obj");
 111   mpc_parser_t* JSON           = mpc_new("json");
 112 
 113   mpca_lang(MPCA_LANG_PREDICTIVE,
 114       " number         : /[0-9]+/                              ; \n"
 115       " key            : /[^'\"''{''}'':'';'' '',']*/          ; \n"
 116       " json_string    : '\"' <key> '\"'                       ; \n"
 117       " json_boolean   : \"true\" | \"false\"                  ; \n"
 118       " kv_item        : <json_string> ':'                         "
 119       "                   (<json_value> | <json_array>)        ; \n"
 120       " json_array     : '[' (<json_value> ','?)* ']'          ; \n"
 121       " json_obj       : '{' (<kv_item> ','?)* '}'             ; \n"
 122       " json_value     : <json_obj> | <json_array>                 "
 123       "                | <json_boolean>                            "
 124       "                | <json_string> | <number>              ; \n"
 125       " json           : /^/ <json_value> /$/                  ; \n",
 126       Number, Key,

 143       Number, Key,
